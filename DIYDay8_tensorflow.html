    return tf.multiply(x,w)+b

In [60]:

定义损失函数¶

In [61]:
def loss(x,y,w,b):
    err=model(x,w,b)-y
    squared=tf.square(err)
    return tf.reduce_mean(squared)

def grad(x,y,w,b):
    with tf.GradientTape() as tape:
        loss_=loss(x,y,w,b)
    return tape.gradient(loss_,[w,b])

def model(x,w,b):
    x=tf.matmul(x,w[0])+b[0]
    x=tf.nn.relu(x)
    x=tf.matmul(x,w[1])+b[1]
    x=tf.nn.relu(x)
    x=tf.matmul(x,w[2])+b[2]
    pre=tf.nn.softmax(x)
    return pre
